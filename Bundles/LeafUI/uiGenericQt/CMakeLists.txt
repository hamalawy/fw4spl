fwLoadProperties()

    fwGui
    fwRuntime
    fwTools
)

    ${Qt5Core_INCLUDE_DIRS}

    ${Qt5Core_LIBRARIES}
    ${Qt5Gui_LIBRARIES}

    ${Qt5Widgets_LIBRARIES}
)
